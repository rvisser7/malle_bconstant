//Goal: compute orbits for all pi and phi

/* Input: given an embedding problem, output an integer*/

G:=WreathProduct(CyclicGroup(3),CyclicGroup(2));
T:=Gpiphi(G);
ebp:=T[1];

/*Prepare Smin*/

//use discriminant
ind:=function(g)
    C:=CycleStructure(g);
    k:=0;n:=0;
    for i:=1 to #C do
       k:=k+C[i][2];
       n:=n+C[i][2]*C[i][1];
    end for;
    return n-k;
end function;
MinIndex:=function(G)
    A:=[];a:=#G;
    for g in G do
        if g ne Id(G) and ind(g) lt a then
            a:=ind(g);
            A:=[g];
        elif g ne Id(G) and ind(g) eq a then
            Append(~A,g);
        end if;
    end for;
    return a,A;
end function;
a,Smin:=MinIndex(G);

//using product of ramified primes
Smin:=[];
for x in G do
    if Order(x) gt 1 then Include(~Smin, x);end if;
end for;


//compute orbits function
//Input: ebp, Smin
//Output: #Sminpi, b(pi,phi)
//
//This version keeps the old function name bpiphi,
//but computes orbits using Magma's Orbits function instead of Burnside.
//It should be faster than summing fixed points over every element of CGfiber.

bpiphi := function(ebp, Smin)

    B   := ebp`B;
    G   := ebp`G;
    C   := ebp`C;
    f   := ebp`f;
    pi  := ebp`pi;
    phi := ebp`phi;

    //Compute Smin ∩ Ker(pi)
    Sminpi := [];
    N := Kernel(pi);

    for s in Smin do
        if s in N then
            Append(~Sminpi, s);
        end if;
    end for;

    if #Sminpi eq 0 then
        return 0, 0;
    end if;

    /*
        Index Sminpi.

        We will build a permutation action on the finite set Sminpi.
        The point i corresponds to Sminpi[i].
    */

    index := AssociativeArray();

    for i := 1 to #Sminpi do
        index[Sminpi[i]] := i;
    end for;

    /*
        Construct the fiber product CGfiber as before.

        C is a GrpAb, so we first make a permutation copy CP of C.
        r : CP -> C is the inverse of the coset action map.
    */

    H := sub< C | Id(C) >;
    r, CP := CosetAction(C, H);
    r := Inverse(r);

    CG, into, onto := DirectProduct(G, CP);

    S := [];

    for g in CG do
        x := onto[1](g);
        aCP := onto[2](g);
        aC := r(aCP);

        if pi(x) eq phi(aC) then
            Include(~S, g);
        end if;
    end for;

    CGfiber := sub< CG | S >;

    /*
        Convert generators of CGfiber into permutations of Sminpi.

        The action is

            (x,a).s = x^(-1) * s^a * x,

        where a is viewed as an integer via f.
    */

    perms := [];

    for h in Generators(CGfiber) do

        x := onto[1](h);
        aCP := onto[2](h);
        aC := r(aCP);
        aInt := IntegerRing()!f(aC);

        images := [];

        for s in Sminpi do
            t := x^(-1) * s^aInt * x;
            Append(~images, index[t]);
        end for;

        Append(~perms, Sym(#Sminpi)!images);

    end for;

    /*
        If CGfiber has no generators, the action is trivial.
        This should only occur in degenerate cases, but we handle it.
    */

    if #perms eq 0 then
        return #Sminpi, #Sminpi;
    end if;

    /*
        Let P be the permutation group induced by CGfiber on Sminpi.
        Then b(pi,phi) is the number of orbits.
    */

    P := sub< Sym(#Sminpi) | perms >;

    O := Orbits(P);

    return #Sminpi, #O;

end function;
